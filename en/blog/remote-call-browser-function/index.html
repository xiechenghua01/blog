    const ws = new WebSocket('ws://127.0.0.1:8080')

    ws.on('open', () => {
      let jsonStr = JSON.stringify({
        type: 'getPasswordEnc',
        value: password,
      })

    })

    ws.on('message', message => {
      let json = JSON.parse(message)

          ws.close()
          resolve(value)
          break
      }
    })
  })
}
